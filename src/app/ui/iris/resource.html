iris.resource(
	function(self){

		self.getAll = function(f_ok, f_error){
			return self.get('/##acro##', f_ok, f_error);
		};

		self.create = function( p_##acro##, f_ok, f_error) {
			return self.put('/##acro##', p_##acro##, f_ok);
		};

		self.getFromId = function(p_##acro##Id, f_ok, f_error){
			return self.get('/##acro##/' + p_##acro##Id, f_ok, f_error);
		};

		self.updateFromId = function(p_##acro##Id, p_##acro##, f_ok, f_error){
			return self.post('/##acro##/' + p_##acro##Id, p_##acro##, f_ok, f_error);
		};

		self.removeFromId = function(p_##acro##Id, f_ok, f_error) {
			return self.del('/##acro##/' + p_##acro##Id, f_ok, f_error);
		};

		self.generate = function(p_##acro##Id, p_field, f_ok, f_error){
			return self.post('/##acro##/' + p_##acro##Id + '/generate/' + p_field, {}, f_ok, f_error);
		};

		self.signin = function(p_usr, p_pwd, f_ok, f_error){
			return self.post('/signin', {u: p_usr,p: p_pwd}, function(){
				iris.notify(iris.evts.usr.signing);
				f_ok();
			}, f_error);
		};

	},
	iris.path.resource.##acro##
);
